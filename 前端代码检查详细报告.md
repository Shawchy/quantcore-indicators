### 6.2 WebSocket 服务 (websocket.ts) ✅

```typescript
class WebSocketService {
  private ws: WebSocket | null = null;
  private reconnectTimer: number | null = null;
  private heartbeatTimer: number | null = null;
  private messageHandlers: Map<string, Set<MessageHandler>> = new Map();
  
  // 配置
  private readonly reconnectDelay = 3000;
  private readonly maxReconnectDelay = 30000;
  private readonly heartbeatInterval = 30000;
  private readonly connectionTimeout = 10000;

  connect(token?: string): Promise<void> {
    const url = token ? `${this.url}?token=${token}` : this.url;
    this.ws = new WebSocket(url);
    
    this.ws.onopen = () => {
      this.stats.isConnected = true;
      this.startHeartbeat();
      this.resubscribeAll();
    };
    
    this.ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      this.handleMessage(message);
    };
  }

  subscribe(topic: string): Promise<void> {
    const message = { type: 'system', event: 'subscribe', data: { topic } };
    this.send(message);
    this.stats.subscriptions.add(topic);
  }

  private scheduleReconnect(token?: string) {
    const delay = Math.min(
      this.reconnectDelay * Math.pow(2, this.stats.reconnectCount),
      this.maxReconnectDelay
    );
    
    setTimeout(() => this.connect(token), delay);
  }
}
```

```typescript
class ErrorBoundary extends Component<Props, State> {
  public static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  public componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('ErrorBoundary caught an error:', error, errorInfo)
  }

  public render() {
    if (this.state.hasError) {
      return (
        <Box display="flex" alignItems="center" justifyContent="center" minH="100vh">
          <VStack spacing={4} textAlign="center">
            <Heading size="lg" color="red.500">出错了</Heading>
            <Text color="gray.500">{this.state.error?.message}</Text>
            <Button colorScheme="brand" onClick={this.handleReset}>返回首页</Button>
          </VStack>
        </Box>
      )
    }
    return this.props.children
  }
}
```
